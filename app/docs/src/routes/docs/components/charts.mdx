## Usage

```tsx
import { LineChart } from "~/components/ui/charts"
```

```tsx
const chartData = {
  labels: ["January", "February", "March", "April", "May"],
  datasets: [
    {
      label: "Sales",
      data: [50, 60, 70, 80, 90]
    }
  ]
}
<LineChart data={chartData} />
```
